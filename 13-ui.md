## The key rule for the paywall screen

Do NOT load offerings in `initState`. RC may not be initialized yet when the screen first builds.

Instead, use a `BlocListener` that fires when the BLoC transitions to `ready`. This guarantees you only fetch data after RC is configured.

## Paywall screen

```dart
class SubscriptionsScreen extends StatefulWidget {
  const SubscriptionsScreen({super.key});

  @override
  State<SubscriptionsScreen> createState() => _SubscriptionsScreenState();
}

class _SubscriptionsScreenState extends State<SubscriptionsScreen> {
  @override
  Widget build(BuildContext context) {
    return MultiBlocListener(
      listeners: [
        // Load offerings the moment RC becomes ready
        // listenWhen ensures this fires exactly once: when transitioning to ready
        BlocListener<SubscriptionsBloc, SubscriptionState>(
          listenWhen: (prev, curr) =>
              prev.status != SubscriptionStatus.ready &&
              curr.status == SubscriptionStatus.ready &&
              curr.offerings.isEmpty,
          listener: (context, state) {
            context.read<SubscriptionsBloc>()
              ..add(LoadOfferings())
              ..add(LoadEntitlements());
          },
        ),
        // Show errors and success messages as snack bars
        BlocListener<SubscriptionsBloc, SubscriptionState>(
          listenWhen: (p, c) =>
              c.status == SubscriptionStatus.success ||
              c.status == SubscriptionStatus.error,
          listener: (context, state) {
            final isError = state.status == SubscriptionStatus.error;
            ScaffoldMessenger.of(context).showSnackBar(
              SnackBar(
                content: Text(
                  isError
                      ? (state.errorMessage ?? 'Something went wrong')
                      : (state.successMessage ?? 'Done'),
                ),
                backgroundColor: isError ? Colors.red : Colors.green,
                behavior: SnackBarBehavior.floating,
              ),
            );
          },
        ),
      ],
      child: Scaffold(
        appBar: AppBar(title: const Text('Choose a plan')),
        body: BlocBuilder<SubscriptionsBloc, SubscriptionState>(
          builder: (context, state) {
            // Show shimmer/loading while RC initializes
            if (state.status == SubscriptionStatus.loading &&
                state.offerings.isEmpty) {
              return const Center(child: CircularProgressIndicator());
            }

            // Show empty state if no offerings came back
            if (state.offerings.isEmpty) {
              return const Center(
                child: Text('No plans available. Please try again later.'),
              );
            }

            // Show the plans
            return ListView.builder(
              padding: const EdgeInsets.all(16),
              itemCount: state.offerings.length,
              itemBuilder: (_, i) {
                final pkg = state.offerings[i];

                // Check if the user already owns this product
                final isOwned = state.entitlements.values.any(
                  (e) => e.isActive &&
                      e.productIdentifier == pkg.productIdentifier,
                );

                return _PlanTile(
                  package: pkg,
                  isOwned: isOwned,
                  isPurchasing: state.status == SubscriptionStatus.purchasing,
                );
              },
            );
          },
        ),
        // Restore button — required by App Store guidelines
        bottomNavigationBar: Padding(
          padding: const EdgeInsets.all(16),
          child: BlocBuilder<SubscriptionsBloc, SubscriptionState>(
            builder: (context, state) {
              return TextButton(
                onPressed: state.status == SubscriptionStatus.restoring
                    ? null
                    : () => context
                          .read<SubscriptionsBloc>()
                          .add(RestorePurchases()),
                child: state.status == SubscriptionStatus.restoring
                    ? const SizedBox(
                        width: 16,
                        height: 16,
                        child: CircularProgressIndicator(strokeWidth: 2),
                      )
                    : const Text('Restore Purchases'),
              );
            },
          ),
        ),
      ),
    );
  }
}

class _PlanTile extends StatelessWidget {
  final RCPackage package;
  final bool isOwned;
  final bool isPurchasing;

  const _PlanTile({
    required this.package,
    required this.isOwned,
    required this.isPurchasing,
  });

  @override
  Widget build(BuildContext context) {
    return Card(
      margin: const EdgeInsets.only(bottom: 12),
      child: ListTile(
        title: Text(package.localizedTitle),
        subtitle: Text(package.localizedPriceString),
        trailing: isOwned
            ? const Icon(Icons.check_circle, color: Colors.green)
            : FilledButton(
                onPressed: isPurchasing
                    ? null
                    : () => context
                          .read<SubscriptionsBloc>()
                          .add(PurchasePackage(package)),
                child: isPurchasing
                    ? const SizedBox(
                        width: 14,
                        height: 14,
                        child: CircularProgressIndicator(strokeWidth: 2),
                      )
                    : const Text('Subscribe'),
              ),
      ),
    );
  }
}
```

## Gating premium features

The simplest pattern — use `state.isPremium` anywhere:

```dart
// In a build method:
BlocBuilder<SubscriptionsBloc, SubscriptionState>(
  builder: (context, state) {
    if (state.isPremium) {
      return const PremiumFeatureWidget();
    }
    return const UpgradePromptWidget();
  },
)
```

Imperatively (inside a button handler etc.):

```dart
final isPremium = context.read<SubscriptionsBloc>().state.isPremium;

if (!isPremium) {
  // Navigate to paywall
  context.router.push(const SubscriptionsRoute());
  return;
}

// User is premium — proceed
doThePremiumThing();
```

## Checking a specific entitlement

If you have multiple entitlements (e.g. `premium` and `pro`):

```dart
final state = context.read<SubscriptionsBloc>().state;

final hasPremium = state.entitlements['premium']?.isActive == true;
final hasPro = state.entitlements['pro']?.isActive == true;
```
